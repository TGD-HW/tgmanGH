428  xFunctionPointer getFuncAddress(TGM_LPCTSTR aFileName, TGM_LPCSTR funcName)
429  {
430  assert(aFileName != nullptr);
431  assert(funcName != nullptr);
432  xFunctionPointer func = m_hModule.resolve(funcName);
433  if (!func)
434  {
435  errFuncNotFound(aFileName, funcName);
436  }
437  return func;
438  }
439  bool preLoadDll(TGM_LPCTSTR aFileName)
440  {
441  assert(aFileName != nullptr);
442  CT2A_Converter tFileName(aFileName);
443  m_hModule.setFileName(tFileName.c_str());
444  m_hModule.load();
445  if (m_hModule.isLoaded() == false)
446  {
447  errDllNotFound(aFileName);
448  return false;
449  }
450  return true;
451  }
